\begin{frame}[fragile]
\frametitle{Polling the buffer}
\begin{lstlisting}[frame=trBL,caption={}]{hallowelt}
while (true) {
  if (queue.pop(object)) {
    sleepMicros = 1; // slow start
    doSomething(object);
  } else {
    // The queue is empty! 
    usleep(sleepMicros);
    if (sleepMicros < 1E4) {
      sleepMicros *= 2;
    }
  }
}			
\end{lstlisting}
\end{frame}
